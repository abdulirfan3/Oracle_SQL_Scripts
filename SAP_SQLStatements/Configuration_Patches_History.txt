SELECT NULL ACTION_TIME, NULL ACTION, NULL NAMESPACE, NULL VERSION,
  NULL BUNDLE_SERIES, NULL COMMENTS FROM DUAL WHERE 1 = 0
UNION ALL (
SELECT NULL ACTION_TIME, NULL ACTION, NULL NAMESPACE, NULL VERSION,
  NULL BUNDLE_SERIES, NULL COMMENTS FROM DUAL WHERE 1 = 0
) UNION ALL ( SELECT * FROM (
WITH BASIS_INFO AS
( SELECT
    '%' BUNDLE_SERIES           /* e.g. SBP */
  FROM
    DUAL
)
SELECT
  TO_CHAR(RH.ACTION_TIME, 'YYYY/MM/DD hh24:MI:SS') ACTION_TIME,
  RH.ACTION,
  RH.NAMESPACE,
  RH.VERSION,
  RH.BUNDLE_SERIES,
  RH.COMMENTS
FROM
  BASIS_INFO BI,
  DBA_REGISTRY_HISTORY RH
WHERE
  NVL(RH.BUNDLE_SERIES, ' ') LIKE BI.BUNDLE_SERIES 
ORDER BY
  ACTION_TIME DESC
))
